function squareRoot(N, Eps) {
    let left = 1
    let right = N
    let center = (left + right) / 2

    let iterations = 0
    while(Math.abs(center * center - N) > Eps) {
        center = (left + right) / 2
        if(center * center < N) {
            left = center
        } else {
            right = center
        }

        iterations++
        if(iterations > 200) return NaN;
    }

    return center
}
